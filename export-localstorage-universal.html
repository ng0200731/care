// Run this in console at http://localhost:3002
const allKeys = Object.keys(localStorage);
console.log('Found keys:', allKeys);

// Find project layouts
const layoutKeys = allKeys.filter(k => k.includes('layout'));
console.log('Layout keys:', layoutKeys);

// Export project layouts
layoutKeys.forEach(key => {
    const data = localStorage.getItem(key);
    console.log('\n========== ' + key + ' ==========');
    console.log(data);
});

// Or export to file
const exportKey = 'project_tcl-2025_layouts'; // Change this if needed
const data = localStorage.getItem(exportKey);
if (data) {
    const blob = new Blob([data], { type: 'application/json' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = exportKey + '.json';
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
    console.log('‚úÖ File downloaded!');
} else {
    console.log('‚ùå Key not found:', exportKey);
    console.log('Available keys:', allKeys);
}
